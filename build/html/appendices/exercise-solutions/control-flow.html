String Practice

  1. For this exercise, create a string for the value.

  2. Use the Split method to divide the string at each space and store the individual words in an array.

  3. Print the array of words to verify that your code works.

1   string sentence = "I would not, could not, in a box. I would not, could not with a fox. I will not eat them in a house. I will not eat them with a mouse.";
2   string[] words = sentence.Split(" ");
3   Console.WriteLine(string.Join("/", words));

List Practice

  1. Write a static method to find the sum of all the even numbers in a List.

 1   static int sumEven(List<int> arr)
 2   {
 3      int total = 0;
 4      foreach (int integer in arr)
 5      {
 6         if (integer % 2 == 0)
 7         {
 8            total += integer;
 9         }
10      }
11      return total;
12   }

  1. Write a static method to print out each word in a list that has exactly 5 letters.

 1   static void printFiveLetterWords(List<string> wordlist)
 2   {
 3      foreach (string word in wordlist)
 4      {
 5         if (word.Length == 5)
 6         {
 7            Console.WriteLine(word);
 8         }
 9      }
10   }

  1. Modify your code to prompt the user to enter the word length for the search.

 1   Console.WriteLine("Enter a word length: ");
 2   string numInput = Console.ReadLine();
 3   int numChars = int.Parse(numInput);
 4
 5   // Call the method to print out list words of the chosen length:
 6   printXLetterWords(wordList, numChars);
 7
 8
 9   static void printXLetterWords(List<string> wordlist, int length)
10   {
11      foreach (string word in wordlist)
12      {
13         if (word.Length == length)
14         {
15            Console.WriteLine(word);
16         }
17      }
18   }

Dictionary Practice

  1. It takes in student names and ID numbers (as integers) instead of names and grades.

 1   Console.WriteLine("Enter your students' names and ID numbers (or ENTER to finish):");
 2
 3   Console.WriteLine("Student Name: ");
 4   newStudent = Console.ReadLine();
 5
 6   if (newStudent!= "")
 7   {
 8      Console.WriteLine("ID: ");
 9      int newID = int.Parse(Console.ReadLine());
10      students.Add(newID, newStudent);
11   }

  1. The keys should be the IDs and the values should be the names

1   Console.WriteLine("\nClass roster:");
2
3   foreach (KeyValuePair<int, string> student in students)
4   {
5      Console.WriteLine(student.Value + "'s ID: " + student.Key);
6   }
7
8   Console.WriteLine("Number of students in roster: " + students.Count);
